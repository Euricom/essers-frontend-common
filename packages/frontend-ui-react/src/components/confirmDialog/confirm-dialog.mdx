# Confirm Dialog

```jsx
import { useConfirmDialog, DailogProvider } from '@essers/frontend-ui';

import { DailogProvider } from '@essers/frontend-ui-react'
const Root = () => {
  return (
    <DailogProvider>
      <App />
    </DailogProvider>
  );
};

export App = () => (
  const { openConfirmDialog } = useConfirmDialog();

  const handleClick = () => {
    const { t } = useTranslation();
    const isConfirmed = await openConfirmDialog({
      title: t('Confirmation'),
      content: t('Are you sure to delete the product?'),
    });

    console.log(isConfirmed)
  }
);
```
