# Pagination

```jsx
import { Pagination } from '@essers/frontend-ui';

export MyComponent = () => (
  const [page, setPage] = useState(0);
  const handlePageChange = (page: number) => {
    setPage(page);
  };
  return (
    <div>
      <Pagination totalItems={100} pageSize={5} page={page} onPageChange={handlePageChange} />
    </div>
  );
);
```
